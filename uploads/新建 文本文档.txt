function Yprime = isometric(t,Y)
global L0 alpha beta a S0 gamma t0 tip

%-----------------(Write code hereâ€¦)--------------------
Yprime = zeros(2,1);
T=Y(1);
Lc=Y(2);
L=L0;
Ls=L-Lc;

     Tp=beta*(exp(alpha*(L-L0))-1);
     Ts=beta*(exp(alpha*Ls)-1);     
     T=Tp+Ts;
  if (t<=2*tip+t0)&&(t>=0)
      f=sin((pi/2)*((t+t0)/(tip+t0)));
  elseif (t>=2*tip+t0)
      f=0;  
  end
  dLcdt=(a*(Ts-S0*f))/(Ts+gamma*S0);
  dTdt=-alpha*beta*exp(alpha*(L0-Lc))*dLcdt;
  Yprime(1)=dLcdt;
  Yprime(2)=dTdt;
 end
